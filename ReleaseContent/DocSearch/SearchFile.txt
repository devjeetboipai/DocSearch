SELECT RL50_Contracts.ContractLeaseClassificationContractType AS RL50_Contracts_ContractLeaseClassificationContractType, RL50_Contracts.ContractSequenceNumber AS RL50_Contracts_ContractSequenceNumber, RL50_Customers.CustomerCompanyName AS RL50_Customers_CustomerCompanyName, BillToes.Alias AS BillToes_Alias, RL50_Contracts.ContractLeaseLessorYield AS RL50_Contracts_ContractLeaseLessorYield, RL50_Branches.BranchBranchName AS RL50_Branches_BranchBranchName, RL50_Contracts.ContractPostDate AS RL50_Contracts_ContractPostDate, RL50_LeaseAssets.LeaseAssetNBV_Amount AS RL50_LeaseAssets_LeaseAssetNBV_Amount, RL50_Assets.AssetAlias AS RL50_Assets_AssetAlias, RL50_Contracts.ContractCommencementDate AS RL50_Contracts_ContractCommencementDate   FROM (((((((RL50_Contracts RL50_Contracts        ^ INNER ^  JOIN RL50_Customers RL50_Customers ON ( RL50_Contracts.CustomerId = RL50_Customers.CustomerId      ))        ^ INNER ^  JOIN RL50_BillTos RL50_BillTos ON ( RL50_Contracts.BillToId = RL50_BillTos.BillToId      ))        ^ INNER ^  JOIN BillToes BillToes ON ( RL50_BillTos.BillToId = BillToes.Id      ))        ^ INNER ^  JOIN RL50_Branches RL50_Branches ON ( RL50_Branches.BranchId = RL50_Contracts.BranchId      ))        ^ INNER ^  JOIN RL50_LeaseAssets RL50_LeaseAssets ON ( RL50_Contracts.LeaseFinanceId = RL50_LeaseAssets.LeaseFinanceId      ))        ^ INNER ^  JOIN RL50_Assets RL50_Assets ON ( RL50_Assets.AssetId = RL50_LeaseAssets.AssetId      ))        ^ INNER ^  JOIN RL50_LegalEntities RL50_LegalEntities ON ( RL50_Contracts.LegalEntityId = RL50_LegalEntities.LegalEntityId))  WHERE  (    RL50_Contracts.ContractPostDate >= '11-01-2018' AND     RL50_LegalEntities.LegalEntityName LIKE '%DE%' AND     RL50_Contracts.ContractPostDate <= '11-30-2018' )
